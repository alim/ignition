    The #members_list method will parse the membership list of email addresses check them for valid email format.

    # File app/models/group.rb, line 41
def members_list
  if self.members.present?
    email_list = self.members.split
    email_list.each do |email|
      if !email.match(/^.+@.+\..+/)
        self.errors.add(:members, "invalid email address - #{email}")
      end
    end
  end
end
