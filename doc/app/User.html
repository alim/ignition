    class User

    The User model is responsible for holding information associated with a user account. This model is then used by the Devise GEM for user user authentication and sign in. The model has been upgraded to include timestamps and strip_attributes for removing leading and trailing whitespaces.

def self.filter_by(filter)
  case filter
  when 'customer'
    self.by_role(User::CUSTOMER)
  when 'service_admin'
    self.by_role(User::SERVICE_ADMIN)
  else
    self.all
  end
end

def self.search_by(search_type, search_term)
  # Check for the type of search we are doing
  case search_type
  when 'email'
    self.by_email(search_term)
  when 'first_name'
    self.by_first_name(search_term)
  when 'last_name'
    self.by_last_name(search_term)
  else # Unrecognized search type so return all
    self.all
  end
end

def role_str
  case self.role
  when CUSTOMER
    "Customer"
  when SERVICE_ADMIN
    "Service Administrator"
  else
    "Unknown"
  end
end
